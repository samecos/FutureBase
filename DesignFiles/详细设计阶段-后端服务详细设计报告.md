```java
@Repository
public interface ElementDAO {
    
    @Insert("INSERT INTO elements (id, project_id, floor_id, type, name, geometry, properties, created_by, created_at, version) " +
            "VALUES (#{id}, #{projectId}, #{floorId}, #{type}, #{name}, ST_GeomFromGeoJSON(#{geometry}::json), #{properties}::jsonb, #{createdBy}, #{createdAt}, 1)")
    int insert(Element element);
    
    @Select("SELECT id, project_id, floor_id, type, name, ST_AsGeoJSON(geometry) as geometry, properties, created_by, created_at, updated_at, version " +
            "FROM elements WHERE id = #{id} AND deleted = false")
    @Results({
        @Result(property = "geometry", column = "geometry", typeHandler = GeometryTypeHandler.class),
        @Result(property = "properties", column = "properties", typeHandler = JsonbTypeHandler.class)
    })
    Element findById(String id);
    
    @Select("<script>" +
            "SELECT id, project_id, floor_id, type, name, ST_AsGeoJSON(geometry) as geometry, properties, version " +
            "FROM elements " +
            "WHERE project_id = #{projectId} AND deleted = false " +
            "<if test='floorId != null'> AND floor_id = #{floorId} </if> " +
            "<if test='types != null'> AND type IN <foreach collection='types' item='type' open='(' separator=',' close=')'>#{type}</foreach> </if> " +
            "<if test='bbox != null'> AND ST_Intersects(geometry, ST_MakeEnvelope(#{bbox.minX}, #{bbox.minY}, #{bbox.maxX}, #{bbox.maxY}, 4326)) </if> " +
            "ORDER BY created_at DESC " +
            "LIMIT #{limit} OFFSET #{offset}" +
            "</script>")
    List<Element> findByConditions(QueryConditions conditions);
    
    @Update("UPDATE elements SET " +
            "geometry = COALESCE(ST_GeomFromGeoJSON(#{geometry}::json), geometry), " +
            "properties = COALESCE(#{properties}::jsonb, properties), " +
            "name = COALESCE(#{name}, name), " +
            "updated_at = #{updatedAt}, " +
            "updated_by = #{updatedBy}, " +
            "version = version + 1 " +
            "WHERE id = #{id} AND version = #{version} AND deleted = false")
    int updateWithOptimisticLock(Element element);
    
    @Update("UPDATE elements SET deleted = true, updated_at = #{updatedAt} WHERE id = #{id}")
    int softDelete(@Param("id") String id, @Param("updatedAt") LocalDateTime updatedAt);
    
    @Insert("<script>" +
            "INSERT INTO elements (id, project_id, floor_id, type, name, geometry, properties, created_by, created_at, version) VALUES " +
            "<foreach collection='elements' item='e' separator=','>" +
            "(#{e.id}, #{e.projectId}, #{e.floorId}, #{e.type}, #{e.name}, ST_GeomFromGeoJSON(#{e.geometry}::json), #{e.properties}::jsonb, #{e.createdBy}, #{e.createdAt}, 1)" +
            "</foreach>" +
            "</script>")
    int batchInsert(@Param("elements") List<Element> elements);
    
    @Update("<script>" +
            "<foreach collection='operations' item='op' separator=';'>" +
            "<choose>" +

            "</foreach>" +
```

## 3. 业务逻辑层设计

### 3.1 项目服务实现

```java
// ProjectService.java
@Service
@Transactional
@Slf4j
public class ProjectService {
    
    @Autowired
    private ProjectDAO projectDAO;
    
    @Autowired
    private ProjectMemberDAO memberDAO;
    
    @Autowired
    private FloorDAO floorDAO;
    
    @Autowired
    private EventPublisher eventPublisher;
    
    @Autowired
    private CacheManager cacheManager;
    
    @Autowired
    private IdGenerator idGenerator;
    
    public ProjectDTO createProject(CreateProjectRequest request, String userId) {
        // 参数校验
        validateCreateRequest(request);
        
        // 创建项目实体
        Project project = new Project();
        project.setId(idGenerator.generate());
        project.setName(request.getName());
        project.setDescription(request.getDescription());
        project.setProjectType(request.getProjectType());
        project.setBuildingArea(request.getBuildingArea());
        project.setFloors(request.getFloors());
        project.setStatus(ProjectStatus.DRAFT);
        project.setCreatedBy(userId);
        project.setCreatedAt(LocalDateTime.now());
        project.setUpdatedAt(LocalDateTime.now());
        
        // 保存项目
        projectDAO.insert(project);
        
        // 创建默认楼层
        List<Floor> floors = createDefaultFloors(project.getId(), request.getFloors());
        floorDAO.batchInsert(floors);
        
        // 添加创建者为项目成员
        ProjectMember member = new ProjectMember();
        member.setProjectId(project.getId());
        member.setUserId(userId);
        member.setRole(ProjectRole.OWNER);
        member.setJoinedAt(LocalDateTime.now());
        memberDAO.insert(member);
        
        // 发布项目创建事件
        eventPublisher.publish(new ProjectCreatedEvent(project));
        
        // 缓存项目
        cacheManager.getCache("projects").put(project.getId(), project);
        
        log.info("Project created: {}", project.getId());
        
        return convertToDTO(project);
    }
    
    @Transactional(readOnly = true)
    public PageResult<ProjectDTO> listProjects(ProjectQuery query, String userId) {
        // 获取用户有权限的项目ID列表
        List<String> projectIds = memberDAO.findProjectIdsByUserId(userId);
        
        if (projectIds.isEmpty()) {
            return PageResult.empty();
        }
        
        // 构建查询条件
        QueryConditions conditions = new QueryConditions();
        conditions.setProjectIds(projectIds);
        conditions.setStatus(query.getStatus());
        conditions.setKeyword(query.getKeyword());
        conditions.setSort(query.getSort());
        conditions.setOrder(query.getOrder());
        conditions.setOffset((query.getPage() - 1) * query.getSize());
        conditions.setLimit(query.getSize());
        
        // 查询项目列表
        List<Project> projects = projectDAO.findByConditions(conditions);
        long total = projectDAO.countByConditions(conditions);
        
        // 转换为DTO
        List<ProjectDTO> dtos = projects.stream()
            .map(this::convertToDTO)
            .collect(Collectors.toList());
        
        return PageResult.of(dtos, query.getPage(), query.getSize(), total);
    }
    
    @Transactional(readOnly = true)
    @Cacheable(value = "projects", key = "#projectId")
    public ProjectDTO getProject(String projectId, String userId) {
        // 检查权限
        if (!hasProjectAccess(projectId, userId)) {
            throw new AccessDeniedException("无权限访问该项目");
        }
        
        Project project = projectDAO.findById(projectId);
        if (project == null) {
            throw new NotFoundException("项目不存在");
        }
        
        // 加载楼层信息
        List<Floor> floors = floorDAO.findByProjectId(projectId);
        project.setFloorList(floors);
        
        // 加载成员信息
        List<ProjectMember> members = memberDAO.findByProjectId(projectId);
        project.setMembers(members);
        
        return convertToDTO(project);
    }
    
    public ProjectDTO updateProject(String projectId, UpdateProjectRequest request, String userId) {
        // 检查权限
        if (!hasProjectWriteAccess(projectId, userId)) {
            throw new AccessDeniedException("无权限修改该项目");
        }
        
        Project project = projectDAO.findById(projectId);
        if (project == null) {
            throw new NotFoundException("项目不存在");
        }
        
        // 更新字段
        if (request.getName() != null) {
            project.setName(request.getName());
        }
        if (request.getDescription() != null) {
            project.setDescription(request.getDescription());
        }
        if (request.getBuildingArea() != null) {
            project.setBuildingArea(request.getBuildingArea());
        }
        if (request.getFloors() != null) {
            // 调整楼层数量
            adjustFloors(project, request.getFloors());
        }
        
        project.setUpdatedBy(userId);
        project.setUpdatedAt(LocalDateTime.now());
        
        // 更新项目
        projectDAO.update(project);
        
        // 发布项目更新事件
        eventPublisher.publish(new ProjectUpdatedEvent(project));
        
        // 更新缓存
        cacheManager.getCache("projects").put(projectId, project);
        
        log.info("Project updated: {}", projectId);
        
        return convertToDTO(project);
    }
    
    public void deleteProject(String projectId, String userId) {
        // 检查权限（仅所有者）
        if (!isProjectOwner(projectId, userId)) {
            throw new AccessDeniedException("仅项目所有者可以删除项目");
        }
        
        // 软删除
        projectDAO.softDelete(projectId, LocalDateTime.now());
        
        // 发布项目删除事件
        eventPublisher.publish(new ProjectDeletedEvent(projectId));
        
        // 清除缓存
        cacheManager.getCache("projects").evict(projectId);
        
        log.info("Project deleted: {}", projectId);
    }
    
    // 私有辅助方法
    private void validateCreateRequest(CreateProjectRequest request) {
        if (StringUtils.isBlank(request.getName())) {
            throw new ValidationException("项目名称不能为空");
        }
        if (request.getName().length() < 2 || request.getName().length() > 100) {
            throw new ValidationException("项目名称长度必须在2-100字符之间");
        }
        if (request.getBuildingArea() == null || request.getBuildingArea() <= 0) {
            throw new ValidationException("建筑面积必须大于0");
        }
        if (request.getFloors() == null || request.getFloors() <= 0) {
            throw new ValidationException("楼层数必须大于0");
        }
    }
    
    private List<Floor> createDefaultFloors(String projectId, int floorCount) {
        List<Floor> floors = new ArrayList<>();
        for (int i = 0; i < floorCount; i++) {
            Floor floor = new Floor();
            floor.setId(idGenerator.generate());
            floor.setProjectId(projectId);
            floor.setName(i == 0 ? "首层" : "F" + (i + 1));
            floor.setElevation(i * 3000); // 默认层高3米
            floor.setHeight(3000);
            floor.setSequence(i);
            floors.add(floor);
        }
        return floors;
    }
    
    private void adjustFloors(Project project, int newFloorCount) {
        int currentCount = project.getFloors();
        if (newFloorCount > currentCount) {
            // 增加楼层
            List<Floor> newFloors = new ArrayList<>();
            for (int i = currentCount; i < newFloorCount; i++) {
                Floor floor = new Floor();
                floor.setId(idGenerator.generate());
                floor.setProjectId(project.getId());
                floor.setName("F" + (i + 1));
                floor.setElevation(i * 3000);
                floor.setHeight(3000);
                floor.setSequence(i);
                newFloors.add(floor);
            }
            floorDAO.batchInsert(newFloors);
        } else if (newFloorCount < currentCount) {
            // 减少楼层（需要检查是否有构件）
            List<Floor> floorsToDelete = floorDAO.findByProjectIdAndSequenceGreaterThan(
                project.getId(), newFloorCount - 1);
            
            for (Floor floor : floorsToDelete) {
                // 检查楼层是否有构件
                long elementCount = elementDAO.countByFloorId(floor.getId());
                if (elementCount > 0) {
                    throw new BusinessException("楼层 " + floor.getName() + " 存在构件，无法删除");
                }
                floorDAO.delete(floor.getId());
            }
        }
        project.setFloors(newFloorCount);
    }
    
    private ProjectDTO convertToDTO(Project project) {
        ProjectDTO dto = new ProjectDTO();
        BeanUtils.copyProperties(project, dto);
        
        // 转换楼层
        if (project.getFloorList() != null) {
            dto.setFloors(project.getFloorList().stream()
                .map(this::convertFloorToDTO)
                .collect(Collectors.toList()));
        }
        
        // 转换成员
        if (project.getMembers() != null) {
            dto.setMembers(project.getMembers().stream()
                .map(this::convertMemberToDTO)
                .collect(Collectors.toList()));
        }
        
        return dto;
    }
    
    private boolean hasProjectAccess(String projectId, String userId) {
        return memberDAO.existsByProjectIdAndUserId(projectId, userId);
    }
    
    private boolean hasProjectWriteAccess(String projectId, String userId) {

        return member != null && (member.getRole() == ProjectRole.OWNER || 
                                  member.getRole() == ProjectRole.EDITOR);
    }

        ProjectMember member = memberDAO.findByProjectIdAndUserId(projectId, userId);
```

```java
        if (floor == null || !floor.getProjectId().equals(projectId)) {

        // 检查构件类型是否有效
        if (!isValidElementType(request.getType())) {
            throw new ValidationException("无效的构件类型: " + request.getType());
        }
        
        // 业务规则检查
        if ("wall".equals(request.getType())) {
            checkWallRules(request);
        } else if ("slab".equals(request.getType())) {
            checkSlabRules(request);

    private void checkWallRules(CreateElementRequest request) {
        Map<String, Object> props = request.getProperties();
        if (props != null) {
            Integer thickness = (Integer) props.get("thickness");
            if (thickness != null) {
                String type = (String) props.get("wall_type");
                if ("exterior".equals(type) && thickness < 200) {
                    throw new BusinessException("外墙厚度不得小于200mm");
                } else if ("interior".equals(type) && thickness < 100) {
                    throw new BusinessException("内墙厚度不得小于100mm");
                }
            }
        }
    }

        // 检查楼板闭合性
        Geometry geometry = parseGeometry(request.getGeometry());
        if (!geometry.isClosed()) {
            throw new BusinessException("楼板轮廓必须是闭合多边形");
        }
```

```java
// EventPublisher.java
@Component
@Slf4j
public class EventPublisher {
    
    @Autowired
    private KafkaTemplate<String, String> kafkaTemplate;
    
    @Autowired
    private ObjectMapper objectMapper;
    
    private final Map<Class<?>, List<EventHandler<?>>> handlers = new ConcurrentHashMap<>();
    
    public <T extends DomainEvent> void publish(T event) {
        try {
            // 序列化事件
            String eventJson = objectMapper.writeValueAsString(event);
            
            // 发送到Kafka
            String topic = getTopicForEvent(event.getClass());
            kafkaTemplate.send(topic, event.getAggregateId(), eventJson)
                .addCallback(
                    result -> log.debug("Event published: {}", event.getEventId()),
                    failure -> log.error("Failed to publish event: {}", event.getEventId(), failure)
                );
            
            // 同步调用本地处理器
            handleLocally(event);
            
        } catch (Exception e) {
            log.error("Error publishing event: {}", event, e);
            throw new EventPublishException("事件发布失败", e);
        }
    }
    
    @SuppressWarnings("unchecked")
    private <T extends DomainEvent> void handleLocally(T event) {
        List<EventHandler<?>> eventHandlers = handlers.get(event.getClass());
        if (eventHandlers != null) {
            for (EventHandler<?> handler : eventHandlers) {
                try {
                    ((EventHandler<T>) handler).handle(event);
                } catch (Exception e) {
                    log.error("Error handling event: {} with handler: {}", 
                        event.getEventId(), handler.getClass().getName(), e);
                }
            }
        }
    }
    
    public <T extends DomainEvent> void registerHandler(Class<T> eventType, EventHandler<T> handler) {
        handlers.computeIfAbsent(eventType, k -> new ArrayList<>()).add(handler);
    }
    
    private String getTopicForEvent(Class<?> eventClass) {
        // 根据事件类型路由到不同topic
        if (eventClass.getSimpleName().startsWith("Project")) {
            return "project-events";
        } else if (eventClass.getSimpleName().startsWith("Element")) {
            return "element-events";
        } else if (eventClass.getSimpleName().startsWith("Collaboration")) {
            return "collaboration-events";
        }
        return "domain-events";
    }
}
```

```java
                project, 
                1, TimeUnit.HOURS
```

```java
    // 批量查询优化
    public List<Element> batchQueryOptimized(List<String> elementIds) {
        if (elementIds.isEmpty()) {
            return Collections.emptyList();
        }
        
        // 使用IN查询代替N次单条查询
        String sql = "SELECT * FROM elements WHERE id IN (" + 
                     String.join(",", Collections.nCopies(elementIds.size(), "?")) + 
                     ") AND deleted = false";
        
        return jdbcTemplate.query(sql, elementIds.toArray(), new ElementRowMapper());
    }
    
    // 分页查询优化（游标分页）
    public List<Element> cursorPagination(String projectId, String lastId, int pageSize) {
        String sql = "SELECT * FROM elements " +
                     "WHERE project_id = ? AND deleted = false " +
                     "AND (?::uuid IS NULL OR id > ?::uuid) " +
                     "ORDER BY id " +
                     "LIMIT ?";
        
        return jdbcTemplate.query(sql, 
            ps -> {
                ps.setString(1, projectId);
                ps.setString(2, lastId);
                ps.setString(3, lastId);
                ps.setInt(4, pageSize);
            }, 
            new ElementRowMapper()
        );
    }
    
    // 空间查询优化（使用PostGIS索引）
    public List<Element> spatialQueryOptimized(Envelope bbox) {
        String sql = "SELECT * FROM elements " +
                     "WHERE deleted = false " +
                     "AND ST_Intersects(geometry, ST_MakeEnvelope(?, ?, ?, ?, 4326)) " +
                     "LIMIT 1000";
        
        return jdbcTemplate.query(sql, 
            new Object[]{bbox.getMinX(), bbox.getMinY(), bbox.getMaxX(), bbox.getMaxY()},
            new ElementRowMapper()
        );
    }
}
```

```java
@Service
@Slf4j
public class AsyncService {
    
    @Autowired
    private TaskExecutor taskExecutor;
    
    @Async("taskExecutor")
    public CompletableFuture<Void> processLargeDataset(String projectId) {
        log.info("Starting async processing for project: {}", projectId);
        
        // 执行耗时操作
        processElements(projectId);
        
        return CompletableFuture.completedFuture(null);
    }
    
    @Async("taskExecutor")
    public void generateReportAsync(String projectId, String userId) {
        try {
            Report report = generateReport(projectId);
            
            // 发送通知
            notificationService.sendReportReadyNotification(userId, report);
            
        } catch (Exception e) {
            log.error("Failed to generate report for project: {}", projectId, e);
            notificationService.sendReportFailedNotification(userId, e.getMessage());
        }
    }
}

        executor.setQueueCapacity(200);
        executor.setThreadNamePrefix("async-");
        executor.setRejectedExecutionHandler(new ThreadPoolExecutor.CallerRunsPolicy());

    }
```
