### 1.3 数据一致性策略

#### 1.3.1 一致性级别定义

| 数据类型 | 一致性要求 | 实现策略 | 说明 |
|---------|-----------|---------|------|
| 用户信息 | 强一致性 | 分布式事务 | 用户注册、权限变更必须强一致 |
| 项目元数据 | 强一致性 | 分布式事务 | 项目名称、状态等关键信息 |
| 几何数据 | 最终一致性 | 乐观锁 + 版本向量 | 支持离线编辑和冲突解决 |
| 版本历史 | 强一致性 | 事件溯源 | 版本链必须严格有序 |
| 协作状态 | 最终一致性 | CRDT | 支持实时协作编辑 |
| 分析结果 | 最终一致性 | 异步计算 + 缓存 | 可接受短暂不一致 |
| 日志审计 | 最终一致性 | 异步写入 | 不影响主业务流程 |

#### 1.3.2 分布式事务方案

```sql
-- 使用两阶段提交(2PC)保证强一致性
BEGIN;
  -- 阶段1: 准备
  PREPARE TRANSACTION 'txn_project_create';
  
  -- 插入项目记录
  INSERT INTO projects (id, tenant_id, name, status) 
  VALUES ('proj_001', 'tenant_001', '项目A', 'active');
  
  -- 插入项目成员
  INSERT INTO project_members (project_id, user_id, role)
  VALUES ('proj_001', 'user_001', 'owner');
  
  -- 创建初始版本
  INSERT INTO versions (project_id, version_number, created_by)
  VALUES ('proj_001', 1, 'user_001');
  
  -- 阶段2: 提交
  COMMIT PREPARED 'txn_project_create';
END;
```

```sql
-- 创建空间查询辅助函数

-- 获取指定区域内的几何对象
CREATE OR REPLACE FUNCTION get_geometries_in_bbox(
    p_version_id VARCHAR(64),
    p_min_x FLOAT,
    p_min_y FLOAT,
    p_max_x FLOAT,
    p_max_y FLOAT
) RETURNS TABLE (
    id VARCHAR(64),
    geom_data GEOMETRY,
    properties JSONB
) AS $$
BEGIN
    RETURN QUERY
    SELECT g.id, g.geom_data, g.properties
    FROM geometry g
    WHERE g.version_id = p_version_id
      AND g.is_visible = true
      AND g.is_deleted = false
      AND g.bbox && ST_MakeEnvelope(p_min_x, p_min_y, p_max_x, p_max_y, 4326);
END;
$$ LANGUAGE plpgsql;

-- 获取与指定几何相交的对象
CREATE OR REPLACE FUNCTION get_intersecting_geometries(
    p_version_id VARCHAR(64),
    p_geom GEOMETRY
) RETURNS TABLE (
    id VARCHAR(64),
    geom_data GEOMETRY,
    intersection_area FLOAT
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        g.id, 
        g.geom_data,
        ST_Area(ST_Intersection(g.geom_data, p_geom)) as intersection_area
    FROM geometry g
    WHERE g.version_id = p_version_id
      AND g.is_deleted = false
      AND ST_Intersects(g.geom_data, p_geom);
END;
$$ LANGUAGE plpgsql;

-- 获取指定点半径内的几何对象
CREATE OR REPLACE FUNCTION get_geometries_near_point(
    p_version_id VARCHAR(64),
    p_x FLOAT,
    p_y FLOAT,
    p_radius FLOAT
) RETURNS TABLE (
    id VARCHAR(64),
    geom_data GEOMETRY,
    distance FLOAT
) AS $$
DECLARE
    v_point GEOMETRY;
BEGIN
    v_point := ST_SetSRID(ST_MakePoint(p_x, p_y), 4326);
    
    RETURN QUERY
    SELECT 
        g.id,
        g.geom_data,
        ST_Distance(g.geom_data::geography, v_point::geography) as distance
    FROM geometry g
    WHERE g.version_id = p_version_id
      AND g.is_deleted = false
      AND ST_DWithin(g.geom_data::geography, v_point::geography, p_radius)
    ORDER BY distance;
END;
$$ LANGUAGE plpgsql;
```

```sql
DECLARE
```

#### 6.3.2 权限检查函数

```sql
-- 检查用户权限函数
CREATE OR REPLACE FUNCTION check_user_permission(
    p_user_id VARCHAR(64),
    p_permission_code VARCHAR(100),
    p_resource_type VARCHAR(50) DEFAULT NULL,
    p_resource_id VARCHAR(64) DEFAULT NULL
) RETURNS BOOLEAN AS $$
DECLARE
    v_has_permission BOOLEAN := false;
BEGIN
    -- 1. 检查用户角色权限
    SELECT EXISTS (
        SELECT 1 FROM user_permissions up
        JOIN permissions p ON p.id = up.permission_id
        WHERE up.user_id = p_user_id
          AND p.permission_code = p_permission_code
          AND up.is_granted = true
          AND (up.expires_at IS NULL OR up.expires_at > NOW())
    ) INTO v_has_permission;
    
    IF v_has_permission THEN
        RETURN true;
    END IF;
    
    -- 2. 检查角色权限
    SELECT EXISTS (
        SELECT 1 FROM users u
        JOIN role_permissions rp ON rp.role = u.role
        JOIN permissions p ON p.id = rp.permission_id
        WHERE u.id = p_user_id
          AND p.permission_code = p_permission_code
    ) INTO v_has_permission;
    
    IF v_has_permission THEN
        RETURN true;
    END IF;
    
    -- 3. 检查资源级权限
    IF p_resource_type IS NOT NULL AND p_resource_id IS NOT NULL THEN
        SELECT EXISTS (
            SELECT 1 FROM resource_permissions rp
            WHERE rp.user_id = p_user_id
              AND rp.resource_type = p_resource_type
              AND rp.resource_id = p_resource_id
              AND (rp.expires_at IS NULL OR rp.expires_at > NOW())
        ) INTO v_has_permission;
    END IF;
    
    RETURN v_has_permission;
END;
$$ LANGUAGE plpgsql;
```

#### 6.4.2 配额检查函数

```sql
-- 检查配额函数
CREATE OR REPLACE FUNCTION check_quota(
    p_tenant_id VARCHAR(64),
    p_quota_type quota_type,
    p_increment BIGINT DEFAULT 1
) RETURNS TABLE (
    allowed BOOLEAN,
    current_value BIGINT,
    limit_value BIGINT,
    remaining BIGINT,
    message TEXT
) AS $$
DECLARE
    v_current BIGINT;
    v_limit BIGINT;
    v_remaining BIGINT;
    v_is_hard BOOLEAN;
BEGIN
    -- 获取当前配额使用情况
    SELECT tq.current_value, tq.limit_value, tq.is_hard_limit
    INTO v_current, v_limit, v_is_hard
    FROM tenant_quotas tq
    WHERE tq.tenant_id = p_tenant_id
      AND tq.quota_type = p_quota_type;
    
    -- 如果没有自定义配额，使用套餐默认值
    IF NOT FOUND THEN
        SELECT t.plan_type INTO v_limit
        FROM tenants t WHERE t.id = p_tenant_id;
        
        SELECT pq.limit_value INTO v_limit
        FROM plan_quotas pq
        WHERE pq.plan_type = v_limit
          AND pq.quota_type = p_quota_type;
        
        v_current := 0;
        v_is_hard := true;
    END IF;
    
    -- 无限制
    IF v_limit = -1 THEN
        RETURN QUERY SELECT true, v_current, -1::BIGINT, -1::BIGINT, '无限制'::TEXT;
        RETURN;
    END IF;
    
    v_remaining := v_limit - v_current - p_increment;
    
    IF v_remaining >= 0 THEN
        RETURN QUERY SELECT true, v_current, v_limit, v_remaining, 'OK'::TEXT;
    ELSIF v_is_hard THEN
        RETURN QUERY SELECT false, v_current, v_limit, 0::BIGINT, 
            format('超出配额限制: 当前 %s, 限制 %s', v_current, v_limit);
    ELSE
        RETURN QUERY SELECT true, v_current, v_limit, v_remaining,
            format('警告: 超出软配额限制', v_current, v_limit);
    END IF;
END;
$$ LANGUAGE plpgsql;

-- 更新配额使用函数
CREATE OR REPLACE FUNCTION update_quota_usage(
    p_tenant_id VARCHAR(64),
    p_quota_type quota_type,
    p_delta BIGINT
) RETURNS VOID AS $$
BEGIN
    INSERT INTO tenant_quotas (tenant_id, quota_type, limit_value, current_value)
    VALUES (p_tenant_id, p_quota_type, 0, GREATEST(0, p_delta))
    ON CONFLICT (tenant_id, quota_type)
    DO UPDATE SET 
        current_value = GREATEST(0, tenant_quotas.current_value + p_delta),
        updated_at = NOW();
END;
$$ LANGUAGE plpgsql;
```

```python
class CacheManager:
    def invalidate_project_cache(self, project_id):
        """失效项目相关所有缓存"""
        keys = [
            f"project:meta:{project_id}",
            f"project:members:{project_id}",
            f"project:versions:{project_id}",
        ]
        self.redis.delete(*keys)
        
        # 发布失效消息
        self.redis.publish('cache:invalidation', {
            'pattern': f"project:*:{project_id}*",
            'timestamp': time.time()
        })
    
    def invalidate_version_cache(self, version_id):
        """失效版本相关缓存"""
        # 使用通配符删除
        pattern = f"version:*:{version_id}*"
        for key in self.redis.scan_iter(match=pattern):
            self.redis.delete(key)
```

#### 8.2.2 备份脚本示例

```bash
#!/bin/bash
# 数据库备份脚本

BACKUP_DIR="/backup/postgres"
DATE=$(date +%Y%m%d_%H%M%S)
RETENTION_DAYS=30

# 全量备份
full_backup() {
    echo "Starting full backup at $(date)"
    
    pg_dump -h $DB_HOST -U $DB_USER -d $DB_NAME \
        --format=custom \
        --compress=9 \
        --file="$BACKUP_DIR/full_${DATE}.dump"
    
    # 加密备份
    gpg --encrypt --recipient backup@company.com \
        "$BACKUP_DIR/full_${DATE}.dump"
    
    # 上传到对象存储
    aws s3 cp "$BACKUP_DIR/full_${DATE}.dump.gpg" \
        "s3://backup-bucket/postgres/"
    
    echo "Full backup completed"
}

# 增量备份 (WAL归档)
incremental_backup() {
    echo "Archiving WAL files"
    
    # 强制切换WAL
    psql -h $DB_HOST -U $DB_USER -c "SELECT pg_switch_wal();"
    
    # 归档WAL文件
    rsync -av $PGDATA/pg_wal/ $BACKUP_DIR/wal/
    
    # 上传到对象存储
    aws s3 sync $BACKUP_DIR/wal/ s3://backup-bucket/postgres/wal/
}

# 清理旧备份
cleanup() {
    echo "Cleaning up backups older than $RETENTION_DAYS days"
    
    find $BACKUP_DIR -name "*.dump*" -mtime +$RETENTION_DAYS -delete
    
    # 清理S3旧备份
    aws s3 ls s3://backup-bucket/postgres/ | \
        awk '{print $4}' | \
        xargs -I {} aws s3 rm s3://backup-bucket/postgres/{}
}

# 主逻辑
case "$1" in
    full)
        full_backup
        ;;
    incremental)
        incremental_backup
        ;;
    cleanup)
        cleanup
        ;;
    *)
        echo "Usage: $0 {full|incremental|cleanup}"
        exit 1
        ;;
esac
```
